```
PUT /api/sites/:site/goals/:goalId
```

Updates an existing goal.

### Path Parameters

<TypeTable
  type={{
    goalId: {
      description: 'ID of the goal to update',
      type: 'string | number',
      required: true,
    },
    site: {
      description: 'Site ID or site identifier',
      type: 'string | number',
      required: true,
    },
  }}
/>

### Request Body

<TypeTable
  type={{
    name: {
      description: 'Updated goal name',
      type: 'string',
      required: false,
    },
    goalType: {
      description: 'Type of goal',
      type: 'string',
      typeDescription: '"path" | "event"',
      required: true,
    },
    config: {
      description: 'Updated goal configuration',
      type: 'GoalConfig',
      required: true,
    },
  }}
/>

### Response

<TypeTable
  type={{
    success: {
      description: 'Whether the goal was updated',
      type: 'boolean',
    },
    goalId: {
      description: 'ID of the updated goal',
      type: 'number',
    },
  }}
/>
